### Simulação

<Playground>
{
  () => {
    const [current, setCurrent] = React.useState(0);

    const next = () => setCurrent(current + 1 < 4 ? current + 1 : current);
    const prev = () => setCurrent(current - 1 > 0 ? current - 1 : 0);

    return (
      <>
        <div style={{ display: 'flex', justifyContent: 'center', alignItems: 'center' }}>
          <Button onClick={prev} style={{ margin: '0.5em'}}>Anterior</Button>
          <Button onClick={next} style={{ margin: '0.5em'}}>Próximo</Button>
          <span style={{ padding: '0 0.5em', borderRadius: '5px', color: 'white', backgroundColor: 'black' }}>{current}</span>
        </div>
        <Steps current={current}>
          <Step title="First"/>
          <Step title="Second"/>
          <Step title="Last"/>
        </Steps>
      </>
    );
  }
}
</Playground>
