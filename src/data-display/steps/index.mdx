### Simulação

<Playground>
{
  () => {
    const [current, setCurrent] = React.useState(0);

    const next = () => setCurrent(current + 1);
    const prev = () => setCurrent(current - 1);

    return (
      <>
        <div style={{ display: 'flex', justifyContent: 'center' }}>
          <Button onClick={prev} style={{ margin: '0.5em'}}>Anterior</Button>
          <Button onClick={next} style={{ margin: '0.5em'}}>Próximo</Button>
        </div>
        <Steps current={current}>
          <Step title="First"/>
          <Step title="Second"/>
          <Step title="Last"/>
        </Steps>
      </>
    );
  }
}

</Playground>
